% Bulleted List
\makeatletter
\newcommand{\makelist}[1]{
	\begin{itemize}[noitemsep, topsep=-0.2em, label=\raisebox{0.45ex}{\tiny$\bullet$}] \item#1\makelistchecknextarg}
\newcommand{\makelistchecknextarg}{\@ifnextchar\bgroup{\makelistgobblenextarg}{\end{itemize}}}
\newcommand{\makelistgobblenextarg}[1]{\item#1 \@ifnextchar\bgroup{\makelistgobblenextarg}{\end{itemize}}}
\makeatother

% Split Page
% The first argument is for the left page.
% The rest of the arguments are for the right page (and are separated by a line break).
\makeatletter
\newcommand{\splitpage}[2]{
	\hspace{0.05 \textwidth}
	%
	\begin{minipage}[t]{0.23 \textwidth}
		#1:\strut
	\end{minipage}
	%
	\begin{minipage}[t]{0.7 \textwidth}
		#2\splitpagechecknextarg}
\newcommand{\splitpagechecknextarg}{\@ifnextchar\bgroup{\splitpagegobblenextarg}{\end{minipage}}}
\newcommand{\splitpagegobblenextarg}[1]{\par#1 \@ifnextchar\bgroup{\splitpagegobblenextarg}{\strut\end{minipage}}}
\makeatother
